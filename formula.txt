function absOfDiff(cell2,cell1) {
	var val1 = this.getRefData(cell1);
	var val2 = this.getRefData(cell2);
	return Math.abs(val1 - val2);
}

function countOfPositive(area) {
	var data = this.getRefData(area);
	var ret = 0;
	data.forEach(function (each) {
		if (each[0] > 0) {
			ret += 1;
		}	
	})
	return ret;
}
